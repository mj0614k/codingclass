1부터 100까지 출력하기

1.1 while문을 이용해서 1부터 100까지 출력하기

i = 1;
while(i <= 100) {
    document.write(i)
    i++;
}

1.3 for문을 이용해서 1부터 100까지 짝수만, 홀수만 출력하기

// 짝수만 출력하기
for( let i=2; i<=100; i+=2 ){
    document.write(i)
}

// 홀수만 출력하기
for( let i=1; i<=100; i+=2 ){
    document.write(i)
}

1.4 for문을 이용해서 1부터 100까지의 합 출력하기

let sum = 0;
for(let i=1; i<=100; i++){
    sum = sum + i;  // 첫번째 방법
    sum += i;          // 두번째 방법
}
document.write(sum);
