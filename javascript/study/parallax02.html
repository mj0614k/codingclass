JAVASCRIPT 소스

preventDefault() 클릭 시 페이지 이동 이벤트를 제거합니다.
Attribute() 선택한 요소의 특정 속성 값을 가져옵니다.
scrollIntoView() 특정 요소의 위치로 스크롤을 이동시킵니다.
behavior: smooth 옵션을 추가하면 부드럽게 이동합니다. 
document.querySelectorAll("#parallax__dot a").forEach((el) => {
    el.addEventListener("click", (e) => {
        e.preventDefault();

        document.querySelector(el.getAttribute("href")).scrollIntoView({ behavior: "smooth" });
    });
});

window.addEventListener("scroll", () => {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;

    document.querySelector("#parallax__info span").innerText = Math.floor(scrollTop);
    document.querySelectorAll(".content__item").forEach((e, i) => {
        if (scrollTop >= e.offsetTop - window.innerHeight - 10) {  /* 오차를 줄이기 위해 -10 */
        /* 스크롤의 현재 위치 >= 각 섹션의 위치 - 브라우저의 높이 */
            document.querySelectorAll("#parallax__dot li").forEach((li) => {
                li.classList.remove("active"); /* 각 섹션에 도착하면 active 클래스를 제거 */
            });
            document.querySelector("#parallax__dot li:nth-child(" + (i + 1) + ")").classList.add("active");
        }
    });
});
