레이아웃 : 전체 사이트 유형01

지금까지 카드 유형, 텍스트 유형, 이미지/텍스트 유형 등 다양한 유형의 레이아웃을 작업해 보았습니다. 이번에는 작업했던 유형들을 모두 한 곳에 모아 전체적인 사이트 한 페이지를 만들고, 미디어 쿼리까지 작업해 보도록 하겠습니다.

HTML 소스

HTML 소스는 지금까지 작업한 모든 것들을 한 곳에 모은 것과 같습니다. 헤더 메뉴는 웹에서는 일반적인 메뉴 형태로, 모바일에서는 햄버거 메뉴로 작업해 주었습니다. 각 메뉴를 클릭했을 때, 각각의 위치로 이동할 수 있도록 각 영역은 section 태그로 작업해 주었습니다. 슬라이드 영역을 다시 작업해준 이유는 swiper를 사용하기 위함입니다. 드래그를 통한 이미지 슬라이드 효과를 사용하기 위해 swiper를 이용했고, swiper의 style에 맞추어 클래스를 새롭게 부여해 주었습니다.

자바스크립트 소스는 다음과 같습니다.

//swiper
var swiper = new Swiper(".mySwiper", {
    navigation: {
        nextEl: ".swiper-button-next",
        prevEl: ".swiper-button-prev",
    },
    autoplay: {
        delay: 2000,
        disableOnInteraction: false, /* 슬라이드를 드래그하면 정지되는 버그 제거 */
    },
    pagination: {
        el: ".swiper-pagination",
        type: "bullets",
        clickable: true,
    },
});

// 선택자
const btnStop = document.querySelector(".swiper-button-stop");
const btnStart = document.querySelector(".swiper-button-play");
const btnHam = document.querySelector(".header__ham");
const btnMenu = document.querySelector(".header__menu");

// swiper에서 재생 버튼을 누르면 슬라이드 재생, 재생 버튼 숨김, 정지 버튼 나타남
// swiper에서 정지 버튼을 누르면 슬라이드 정지, 정지 버튼 숨김, 재생 버튼 나타남
btnStart.style.display = "none";
btnStop.addEventListener("click", () => {
    swiper.autoplay.stop();
    btnStart.style.display = "block";
    btnStop.style.display = "none";
});
btnStart.addEventListener("click", () => {
    swiper.autoplay.start();
    btnStart.style.display = "none";
    btnStop.style.display = "block";
});

// 햄버거를 클릭하면 메뉴가 나타나도록 설정
btnHam.addEventListener("click", () => {
    btnHam.classList.toggle("active");
    btnMenu.classList.toggle("overlay");
});

// 패럴랙스 효과
window.addEventListener("scroll", () => {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;
    document.querySelectorAll("section").forEach((el, i) => {
        if (scrollTop > el.offsetTop - 500) {
            document.querySelectorAll(".header__menu ul li").forEach((li) => {
                li.classList.remove("active");
            });
            document.querySelector(".header__menu ul li:nth-child(" + (i + 1) + ")").classList.add("active");
        }
    });
});

// 메뉴 이동 시 부드럽게~
document.querySelectorAll(".header__inner li a").forEach((li) => {
    li.addEventListener("click", (e) => {
        e.preventDefault();
        document.querySelector(li.getAttribute("href")).scrollIntoView({
            behavior: "smooth",
        });
    });
});
